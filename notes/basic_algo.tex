\begin{algorithm}[t]
  \caption{iterative calls}\label{algorithm:basic}
  %
  $a\gets []$ \tcp*[r]{empty sequence of assignments}
  \For{$i\in 1..|D|$}{
    \For{$j\in 1..|D|$}{
      $k\gets 1$\;
      \lWhile{$\lnot\SAT(a\texttt{ ++ }[f'(i,j)\mapsto k])$}{
        $k\gets k + 1$
      }
      $a\gets a\texttt{ ++ }[f'(i,j)\mapsto k]$\tcp*[r]{extend the sequence}
    }
  }
\end{algorithm}
